        // 全局变量
        let currentQ = 0;
        let answers = [];
        let score = { EI: 0, SN: 0, TF: 0, JP: 0 };
        const totalQ = questions.length;

        // 页面切换
        function hideAllPage() {
            document.querySelectorAll(".page").forEach(p => p.classList.add("hidden"));
        }
        function showPage(id) {
            hideAllPage();
            document.getElementById(id).classList.remove("hidden");
        }

        // 开始测试
        function startTest() {
            currentQ = 0;
            answers = [];
            score = { EI: 0, SN: 0, TF: 0, JP: 0 };
            renderQuestion();
            showPage("page-test");
        }

        // 渲染题目
        function renderQuestion() {
            const q = questions[currentQ];
            document.getElementById("question-num").innerText = `第${currentQ + 1}题`;
            document.getElementById("question-text").innerText = q.q;
            // 更新进度
            const per = ((currentQ) / totalQ) * 100;
            document.getElementById("progress-bar").style.width = per + "%";
            document.getElementById("progress-text").innerText = `${currentQ}/${totalQ}`;
            // 重置按钮样式
            document.querySelectorAll(".answer-btn").forEach(btn => {
                btn.classList.remove("border-primary", "bg-light");
            });
        }

        // 选择答案
        function selectAnswer(val) {
            const q = questions[currentQ];
            // 计分：A-E对应5~1分，反向维度取反
            let point = 6 - val;
            if(q.score === "E" || q.score === "S" || q.score === "T" || q.score === "P"){
                point = val - 1;
            }
            score[q.type] += point;
            answers.push(val);
            currentQ++;
            if(currentQ >= totalQ){
                calcResult();
                return;
            }
            renderQuestion();
        }

        // 计算人格结果
        function calcResult() {
            const totalEI = 28;
            const totalSN = 28;
            const totalTF = 28;
            const totalJP = 28;
            let code = "";
            // EI
            code += score.EI > totalEI / 2 ? "I" : "E";
            // SN
            code += score.SN > totalSN / 2 ? "N" : "S";
            // TF
            code += score.TF > totalTF / 2 ? "F" : "T";
            // JP
            code += score.JP > totalJP / 2 ? "P" : "J";
            renderResult(code);
        }

        // 渲染结果页
        function renderResult(code) {
            const data = mbtiData[code];
            document.getElementById("result-code").innerText = code;
            document.getElementById("result-name").innerText = data.name;
            document.getElementById("result-slogan").innerText = data.slogan;
            document.getElementById("result-desc").innerText = data.desc;
            document.getElementById("result-good").innerText = data.good;
            document.getElementById("result-bad").innerText = data.bad;
            document.getElementById("result-job").innerText = data.job;
            document.getElementById("result-love").innerText = data.love;
            // 四维文字
            const eis = score.EI > 14 ? "I 内向" : "E 外向";
            const sns = score.SN > 14 ? "N 直觉" : "S 实感";
            const tfs = score.TF > 14 ? "F 情感" : "T 思考";
            const jps = score.JP > 14 ? "P 感知" : "J 判断";
            document.getElementById("dim-EI").innerText = eis;
            document.getElementById("dim-SN").innerText = sns;
            document.getElementById("dim-TF").innerText = tfs;

        }
